set(LPM_INCLUDE_ARGS
    "${FFX_GPU_PATH}"
    "${FFX_GPU_PATH}/lpm")

file(GLOB LPM_SHADERS_HLSL
    "shaders/lpm/*.hlsl")

# compile all the shaders in HLSL
compile_shaders("${FFX_SC_EXECUTABLE}" OFF "" "" "${LPM_INCLUDE_ARGS}" "${LPM_SHADERS_HLSL}" ${FFX_PASS_SHADER_OUTPUT_PATH_DXIL} LPM_PERMUTATION_OUTPUTS_DXIL)

# add the header files they generate to the main list of dependencies
add_shader_output("${LPM_PERMUTATION_OUTPUTS_DXIL}")

file(GLOB LPM_SHADERS_GLSL
    "shaders/lpm/*.glsl")

# compile all the shaders	
compile_shaders("${FFX_SC_EXECUTABLE}" ON "" "${LPM_PERMUTATION_ARGS}" "${LPM_INCLUDE_ARGS}" "${LPM_SHADERS_GLSL}" ${FFX_PASS_SHADER_OUTPUT_PATH_SPIRV} LPM_PERMUTATION_OUTPUTS_SPIRV)

# add the header files they generate to the main list of dependencies
add_shader_output("${LPM_PERMUTATION_OUTPUTS_SPIRV}")
