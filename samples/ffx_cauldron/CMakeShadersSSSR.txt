set(SSSR_PERMUTATION_ARGS
    -DFFX_SSSR_OPTION_INVERTED_DEPTH={0,1})

set(SSSR_DX12_BASE_ARGS)

set(SSSR_INCLUDE_ARGS
	"${FFX_GPU_PATH}"
	"${FFX_GPU_PATH}/sssr")

file(GLOB SSSR_SHADERS_HLSL
    "shaders/sssr/*.hlsl")

# compile all the shaders in HLSL
compile_shaders("${FFX_SC_EXECUTABLE}" OFF "${SSSR_DX12_BASE_ARGS}" "${SSSR_PERMUTATION_ARGS}" "${SSSR_INCLUDE_ARGS}" "${SSSR_SHADERS_HLSL}" ${FFX_PASS_SHADER_OUTPUT_PATH_DXIL} SSSR_PERMUTATION_OUTPUTS_DXIL)

# add the header files they generate to the main list of dependencies
add_shader_output("${SSSR_PERMUTATION_OUTPUTS_DXIL}")

file(GLOB SSSR_SHADERS_GLSL
    "shaders/sssr/*.glsl")

# compile all the shaders in GLSL
compile_shaders("${FFX_SC_EXECUTABLE}" ON "" "${SSSR_PERMUTATION_ARGS}" "${SSSR_INCLUDE_ARGS}" "${SSSR_SHADERS_GLSL}" ${FFX_PASS_SHADER_OUTPUT_PATH_SPIRV} SSSR_PERMUTATION_OUTPUTS_SPIRV)

# add the header files they generate to the main list of dependencies
add_shader_output("${SSSR_PERMUTATION_OUTPUTS_SPIRV}")
