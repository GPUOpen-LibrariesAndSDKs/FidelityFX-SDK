set(VRS_BASE_ARGS
    -reflection -deps=gcc -DFFX_GPU=1)

set(VRS_CAULDRON_BASE_ARGS
    -E CS -Wno-for-redefinition -Wno-ambig-lit-shift -DFFX_HLSL=1)

set(VRS_PERMUTATION_ARGS
	-DFFX_VRS_OPTION_ADDITIONALSHADINGRATES={0,1}
	-DFFX_VARIABLESHADING_TILESIZE={8,16,32}
	)

set(VRS_INCLUDE_ARGS
	"${FFX_GPU_PATH}"
	"${FFX_GPU_PATH}/vrs")

file(GLOB VRS_SHADERS
    "shaders/vrs/ffx_vrs_imagegen_pass.hlsl")

# compile all the shaders
compile_shaders("${FFX_SC_EXECUTABLE}" "${VRS_BASE_ARGS}" "${VRS_CAULDRON_BASE_ARGS}" "${VRS_PERMUTATION_ARGS}" "${VRS_INCLUDE_ARGS}" "${VRS_SHADERS}" VRS_PERMUTATION_OUTPUTS)

# add the header files they generate to the main list of dependencies
add_shader_output("${VRS_PERMUTATION_OUTPUTS}")
