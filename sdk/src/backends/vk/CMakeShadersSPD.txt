set(SPD_BASE_ARGS
    -reflection -deps=gcc -DFFX_GPU=1)

set(SPD_VK_BASE_ARGS
    -compiler=glslang -e CS --target-env vulkan1.1 -S comp -Os -DFFX_GLSL=1)
	
set(SPD_PERMUTATION_ARGS
	-DFFX_SPD_OPTION_LINEAR_SAMPLE={0,1}
    -DFFX_SPD_OPTION_WAVE_INTEROP_LDS={0,1}
    -DFFX_SPD_OPTION_DOWNSAMPLE_FILTER={0,1,2}
	)
	
set(SPD_INCLUDE_ARGS
	"${FFX_GPU_PATH}"
	"${FFX_GPU_PATH}/spd")

file(GLOB SPD_SHADERS
    "shaders/spd/ffx_spd_downsample_pass.glsl")

# compile all the shaders	
compile_shaders("${FFX_SC_EXECUTABLE}" "${SPD_BASE_ARGS}" "${SPD_VK_BASE_ARGS}" "${SPD_PERMUTATION_ARGS}" "${SPD_INCLUDE_ARGS}" "${SPD_SHADERS}" SPD_PERMUTATION_OUTPUTS)

# add the header files they generate to the main list of dependencies
add_shader_output("${SPD_PERMUTATION_OUTPUTS}")
