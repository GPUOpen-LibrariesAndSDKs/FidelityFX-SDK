set(DOF_BASE_ARGS
    -reflection -deps=gcc -DFFX_GPU=1)

set(DOF_VK_BASE_ARGS
    -compiler=glslang -e CS --target-env vulkan1.1 -S comp -Os -DFFX_GLSL=1)
	
set(DOF_PERMUTATION_ARGS
	-DFFX_DOF_OPTION_MAX_RING_MERGE_LOG={0,1}
    -DFFX_DOF_OPTION_COMBINE_IN_PLACE={0,1}
    -DFFX_DOF_OPTION_REVERSE_DEPTH={0,1}
	)
	
set(DOF_INCLUDE_ARGS
	"${FFX_GPU_PATH}"
	"${FFX_GPU_PATH}/dof")

file(GLOB DOF_SHADERS
    "shaders/dof/ffx_dof_downsample_depth_pass.glsl"
    "shaders/dof/ffx_dof_downsample_color_pass.glsl"
    "shaders/dof/ffx_dof_dilate_pass.glsl"
    "shaders/dof/ffx_dof_blur_pass.glsl"
    "shaders/dof/ffx_dof_composite_pass.glsl")

# compile all the shaders	
compile_shaders("${FFX_SC_EXECUTABLE}" "${DOF_BASE_ARGS}" "${DOF_VK_BASE_ARGS}" "${DOF_PERMUTATION_ARGS}" "${DOF_INCLUDE_ARGS}" "${DOF_SHADERS}" DOF_PERMUTATION_OUTPUTS)

# add the header files they generate to the main list of dependencies
add_shader_output("${DOF_PERMUTATION_OUTPUTS}")
