set(LPM_BASE_ARGS
    -reflection -deps=gcc -DFFX_GPU=1)

set(LPM_VK_BASE_ARGS
    -compiler=glslang -e CS --target-env vulkan1.1 -S comp -Os -DFFX_GLSL=1)
	
set(LPM_INCLUDE_ARGS
	"${FFX_GPU_PATH}"
	"${FFX_GPU_PATH}/lpm")

file(GLOB LPM_SHADERS
    "shaders/lpm/ffx_lpm_filter_pass.glsl")

# compile all the shaders	
compile_shaders("${FFX_SC_EXECUTABLE}" "${LPM_BASE_ARGS}" "${LPM_VK_BASE_ARGS}" "" "${LPM_INCLUDE_ARGS}" "${LPM_SHADERS}" LPM_PERMUTATION_OUTPUTS)

# add the header files they generate to the main list of dependencies
add_shader_output("${LPM_PERMUTATION_OUTPUTS}")
